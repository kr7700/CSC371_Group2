CREATE TABLE ADMINISTRATOR(Name VARCHAR(15) NOT NULL, Password VARCHAR(15) NOT NULL, PRIMARY KEY(Name))
CREATE TABLE BLACKLIST(P_Name VARCHAR(15) NOT NULL, A_Name VARCHAR(15), PRIMARY KEY(P_Name), CONSTRAINT BLKADMFK FOREIGN KEY(A_Name) REFERENCES ADMINISTRATOR(Name) ON DELETE SET NULL ON UPDATE CASCADE)
CREATE TABLE CARTEL(Name VARCHAR(25) NOT NULL, Leader VARCHAR(15) NOT NULL, Resource_Amt INT(10), PRIMARY KEY(Name))
CREATE TABLE CHAT_MESSAGES(C_Name VARCHAR(25) NOT NULL, Message VARCHAR(280), PRIMARY KEY(C_Name, Message), CONSTRAINT CMSGCRTFK FOREIGN KEY(C_Name) REFERENCES CARTEL(Name) ON DELETE CASCADE ON UPDATE CASCADE)
CREATE TABLE BOARD_MESSAGES(C_Name VARCHAR(25) NOT NULL, Message VARCHAR(280), PRIMARY KEY(C_Name, Message), CONSTRAINT BMSGCRTFK FOREIGN KEY(C_Name) REFERENCES CARTEL(Name) ON DELETE CASCADE ON UPDATE CASCADE)
CREATE TABLE FLEET(ID CHAR(3) NOT NULL, Location CHAR(12) NOT NULL, PRIMARY KEY(ID))
CREATE TABLE ORDERS(F_ID CHAR(3) NOT NULL, Move BIT, Location CHAR(12), Load_Resources BIT, Unload_Resources BIT, Resource_Amt INT(10), PRIMARY KEY(F_ID), CONSTRAINT ORDFLEFK FOREIGN KEY(F_ID) REFERENCES FLEET(ID) ON DELETE CASCADE ON UPDATE CASCADE)
CREATE TABLE PLAYER(Name VARCHAR(15) NOT NULL, Password VARCHAR(15) NOT NULL, Money INT(10) NOT NULL, C_Name VARCHAR(25), F_ID CHAR(3), PRIMARY KEY(Name), CONSTRAINT PLACARFK FOREIGN KEY(C_Name) REFERENCES CARTEL(Name) ON DELETE SET NULL ON UPDATE CASCADE, CONSTRAINT PLAFLEFK FOREIGN KEY(F_ID) REFERENCES FLEET(ID) ON DELETE SET NULL ON UPDATE CASCADE)
CREATE TABLE MANAGES(A_Name VARCHAR(15) NOT NULL, P_Name VARCHAR(15) NOT NULL, PRIMARY KEY(A_Name, P_Name), CONSTRAINT MANADMFK FOREIGN KEY(A_Name) REFERENCES ADMINISTRATOR(Name) ON DELETE CASCADE ON UPDATE CASCADE, CONSTRAINT MANPLAFK FOREIGN KEY(P_Name) REFERENCES PLAYER(Name) ON DELETE CASCADE ON UPDATE CASCADE)
CREATE TABLE PRIVATE_MESSAGES(P_Name VARCHAR(15) NOT NULL, Message VARCHAR(280), PRIMARY KEY(P_Name, Message), CONSTRAINT PMSGPLRFK FOREIGN KEY(P_Name) REFERENCES PLAYER(Name) ON DELETE CASCADE ON UPDATE CASCADE)
CREATE TABLE PLANET(ID CHAR(5) NOT NULL, Coordinates CHAR(12) NOT NULL, Bauble_Amt INT(10), Resource_Amt INT(10), P_Name VARCHAR(15), PRIMARY KEY(ID), CONSTRAINT PLNPLAFK FOREIGN KEY(P_Name) REFERENCES PLAYER(Name) ON DELETE SET NULL ON UPDATE CASCADE)
CREATE TABLE RESEARCH_CENTER(ID CHAR(3) NOT NULL, Cost INT(10) NOT NULL, Maintenance_Cost INT(10) NOT NULL, P_ID CHAR(5) NOT NULL, PRIMARY KEY(ID), CONSTRAINT RSCPLNTFK FOREIGN KEY(P_ID) REFERENCES PLANET(ID) ON DELETE CASCADE ON UPDATE CASCADE)
CREATE TABLE RESEARCH(Engine_Level INT(2) NOT NULL, Mining_Level INT(2) NOT NULL, Factory_Level INT(2) NOT NULL, Hull_Level INT(2) NOT NULL, Weapons_Level INT(2) NOT NULL, RC_ID CHAR(3) NOT NULL, PRIMARY KEY(RC_ID), CONSTRAINT RESRSCFK FOREIGN KEY(RC_ID) REFERENCES RESEARCH_CENTER(ID) ON DELETE CASCADE ON UPDATE CASCADE)
CREATE TABLE FACTORY(ID CHAR(3) NOT NULL, Cost INT(10) NOT NULL, Maintenance_Cost INT(10) NOT NULL, Bauble_Alloc INT(10) NOT NULL, P_ID CHAR(5) NOT NULL, PRIMARY KEY(ID), CONSTRAINT FACPLNTFK FOREIGN KEY(P_ID) REFERENCES PLANET(ID) ON DELETE CASCADE ON UPDATE CASCADE)
CREATE TABLE MINE(ID CHAR(3) NOT NULL, Cost INT(10) NOT NULL, Maintenance_Cost INT(10) NOT NULL, P_ID CHAR(5) NOT NULL, PRIMARY KEY(ID), CONSTRAINT MINEPLNTFK FOREIGN KEY(P_ID) REFERENCES PLANET(ID) ON DELETE CASCADE ON UPDATE CASCADE)
CREATE TABLE SHIPYARD(ID CHAR(3) NOT NULL, Cost INT(10) NOT NULL, Maintenance_Cost INT(10) NOT NULL, Cargo_Alloc INT(10) NOT NULL, Cruiser_Alloc INT(10) NOT NULL, P_ID CHAR(5) NOT NULL, PRIMARY KEY(ID), CONSTRAINT SHPYDPLNTFK FOREIGN KEY(P_ID) REFERENCES PLANET(ID) ON DELETE CASCADE ON UPDATE CASCADE)
CREATE TABLE SHIP(P_Name VARCHAR(15) NOT NULL, SY_ID CHAR(3) NOT NULL, ID CHAR(5) NOT NULL, Type VARCHAR(15) NOT NULL, Weapon_Sys INT(2), Resource_Amt INT(4), F_ID CHAR(3) NOT NULL, PRIMARY KEY(ID), CONSTRAINT SHPPLAFK FOREIGN KEY(P_NAME) REFERENCES PLAYER(Name) ON DELETE CASCADE ON UPDATE CASCADE, CONSTRAINT SHPSHPYDFK FOREIGN KEY(SY_ID) REFERENCES SHIPYARD(ID) ON DELETE NO ACTION ON UPDATE CASCADE, CONSTRAINT SHPFLTFK FOREIGN KEY(F_ID) REFERENCES FLEET(ID) ON DELETE RESTRICT ON UPDATE CASCADE)
